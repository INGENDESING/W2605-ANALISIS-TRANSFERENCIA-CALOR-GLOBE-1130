% Este archivo replica la disposición visual del encabezado del informe de
% tercero, pero reemplaza los datos propios de ese documento por las variables

%
% Uso: incluir en el preámbulo del documento LaTeX con:

% Asegurar que los paquetes necesarios estén cargados
\RequirePackage{fancyhdr}
\RequirePackage{tabularx}
\RequirePackage{array}
\RequirePackage{graphicx}
\RequirePackage{multirow}
\RequirePackage{calc}

% Ajustar la altura del encabezado para evitar advertencias de fancyhdr.
% El valor se calcula para 4 filas de \scriptsize (~9 pt cada una) más el
% espacio de las reglas horizontales.
\setlength{\headheight}{100pt}
\addtolength{\topmargin}{-10pt}

% Comando principal del membrete estilo tercero
% Diseño: 4 filas y 5 columnas que ocupan el ancho total de la página
% (\textwidth). Las columnas de etiqueta y el logo tienen anchos fijos;
% las columnas de valor se expanden automáticamente.
%
%   Fila 1: PROJECT | \documentcode          | ~ | Page X/Y  | logo
%   Fila 2: JOB     | Updated                | ~ | \documentdate | logo
%   Fila 3: VERSION | \documentrevision      | Reviewed | \documentdate | logo
%   Fila 4: DESIGNER| \elaboratedby          | Reviewer | \reviewedby | logo
%
% Nota: las etiquetas "Updated", "Reviewed" y "Reviewer" se mantienen en inglés
% para respetar el formato del informe de tercero.
\newcommand{\membreteestilotercero}{%
    \noindent
    \begin{tabularx}{\textwidth}{!{\vrule width 1pt}>{\raggedright\arraybackslash}p{1.6cm} X >{\raggedright\arraybackslash}p{1.7cm} X >{\centering\arraybackslash}p{3.0cm}!{\vrule width 1pt}}
        \noalign{\hrule height 1pt}
        \textbf{\scriptsize PROJECT}  & \textbf{\scriptsize \documentcode} & \rule{0pt}{9pt} & \scriptsize Page \thepage\ de \pageref{LastPage} & \multirow{4}{=}{\centering\includegraphics[width=2.8cm,keepaspectratio]{logos/logo1.png}} \\
        \textbf{\scriptsize JOB}      & \scriptsize Updated & \rule{0pt}{9pt} & \scriptsize \documentdate & \rule{0pt}{9pt} \\
        \textbf{\scriptsize VERSION}  & \textbf{\scriptsize \documentrevision} & \scriptsize Reviewed & \scriptsize \documentdate & \rule{0pt}{9pt} \\
        \textbf{\scriptsize DESIGNER} & \scriptsize \elaboratedby & \scriptsize Reviewer & \scriptsize \reviewedby & \rule{0pt}{9pt} \\
        \noalign{\hrule height 1pt}
    \end{tabularx}%
}

% Estilo de página personalizado
\fancypagestyle{tercerocorporativo}{%
    \fancyhf{} % Limpia encabezados y pies por defecto
    \fancyhead[C]{\membreteestilotercero}
    \renewcommand{\headrulewidth}{0pt} % Sin línea horizontal adicional
    \renewcommand{\footrulewidth}{0pt} % Sin línea en el pie
}

% Aplicar el estilo globalmente si se incluye este archivo
\pagestyle{tercerocorporativo}

% Forzar que los estilos plain, empty, headings y corporativo también usen el membrete
\makeatletter
\let\ps@plain\ps@tercerocorporativo
\let\ps@empty\ps@tercerocorporativo
\let\ps@headings\ps@tercerocorporativo
\let\ps@myheadings\ps@tercerocorporativo
\let\ps@corporativo\ps@tercerocorporativo
\makeatother
